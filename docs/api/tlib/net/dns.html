163    def query(self, verbose: bool = False) -> None:
164        """Send DNS Query. If nothing is found, NoAnswer is raised."""
165        ans = dns.resolver.resolve(self.url, self.rdata_type.value)
166        if verbose:
167            print(f"qname:{ans.qname}, size:{len(ans)}")
168        if self.rdata_type == RDataType.TXT:
169            for rdata in ans:
170                for rdata_str in rdata.strings:
171                    if verbose:
172                        print(rdata_str)
173                    self._txt = str(rdata_str, encoding='utf-8')
174
175        elif self.rdata_type == RDataType.SOA:
176            for rdata in ans:
177                soa = SOA(
178                    mname=str(rdata.mname),
179                    rname=str(rdata.rname),
180                    serial=rdata.serial,
181                    refresh=rdata.refresh,
182                    retry=rdata.retry,
183                    expire=rdata.expire,
184                    minimum=rdata.minimum
185                )
186                if verbose:
187                    print(soa)
188            self._soa = soa
189
190        elif self.rdata_type == RDataType.NAMESERVER:
191            ns_list = []
192            for rdata in ans:
193                if verbose:
194                    print(str(rdata.target))
195                ns_list.append(str(rdata.target))
196            self._nameservers = ns_list
197
198        elif self.rdata_type == RDataType.CNAME:
199            cname_list = []
200            for rdata in ans:
201                if verbose:
202                    print(str(rdata.target))
203                cname_list.append(str(rdata.target))
204            self._canonial_names = cname_list
205
206        elif self.rdata_type == RDataType.ADDRESS:
207            addr_list = []
208            for rdata in ans:
209                if verbose:
210                    print(str(rdata.address))
211                addr_list.append(str(rdata.address))
212            self._addresses = addr_list
213
214        elif self.rdata_type == RDataType.IPV6_ADDRESS:
215            addr_list = []
216            for rdata in ans:
217                if verbose:
218                    print(str(rdata.address))
219                addr_list.append(str(rdata.address))
220            self._addresses_v6 = addr_list
221
222        elif self.rdata_type == RDataType.MAIL_EXCHANGE:
223            mx_list = []
224            for rdata in ans:
225                mx = MXRecord(rdata.preference, rdata.exchange)
226                if verbose:
227                    print(mx)
228                mx_list.append(mx)
229            self._mailexchanges = mx_list
230
231        elif self.rdata_type == RDataType.HOST_INFO:
232            host_info = []
233            for rdata in ans:
234                if verbose:
235                    print(rdata.to_text())
236                host_info.append(rdata.to_text())
237            self._host_information = host_info
238
239        elif self.rdata_type == RDataType.POINTER:
240            pointers = []
241            for rdata in ans:
242                if verbose:
243                    print(rdata.target)
244                pointers.append(rdata.target)
245            self._pointers = pointers
246
247        elif self.rdata_type == RDataType.SERVICE:
248            raise NotImplementedError()
249
250        self._qname = str(ans.qname)
251        self._ans_size = len(ans)
