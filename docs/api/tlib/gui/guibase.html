 41def from_widget_state_to_bool(s: str) -> bool:
 42    """
 43    Convert widget state to boolean representation
 44    NORMAL and ACTIVE => True, else => False
 45    """
 46    if s == WidgetState.NORMAL.value or s == WidgetState.ACTIVE.value:
 47        return True
 48    else:
 49        return False
 50
 51
 52def from_bool_to_widget_state(b: bool) -> WidgetState:
 53    """
 54    Convert bool to widget state
 55    True => ACTIVE, False => DISABLED
 56    """
 57    if b:
 58        return WidgetState.ACTIVE
 59    else:
 60        return WidgetState.DISABLED

532    def build(self):
533        """Build a GUI"""
534        self.root.title(self.title)
535        self.root.geometry(self._to_geometry())
536        if self.menubar is not None:
537            for label, menu in self.menus.items():
538                self.menubar.add_cascade(
539                    label=label,
540                    menu=menu
541                )
542            self.root["menu"] = self.menubar
543        if not self.is_grid_enabled:
544            [w.pack() for n, w in self.widgets.items() if not n.endswith("_val")]
545        else:
546            for _, frame in self.frame.items():
547                frame.frame.pack(padx=frame.padx, pady=frame.pady)
548        return self.root
