258def affine_transform(src: MatLike, scale: int, angle: int) -> MatLike:
259    """Apply affine transformation"""
260    h, w = src.shape[:2]
261    center = (w // 2, h // 2)
262    # M is affin trans matrix
263    M = cv2.getRotationMatrix2D(center, angle, scale)
264    # M = [a    b   c]
265    #     [d    e   f]
266    # a .. scale up/down against X axis. scale * cos(angle)
267    # e .. scale up/down against Y axis. scale * sin(angle)
268    # b .. shear/rotate against x axis
269    # d .. shear/rotate against y axis
270    # c .. parallel move against x axis. (1-a)*center.x - b*center.y
271    # f .. parallel move against y axis. b*center.x - (1-a)*center.y
272    cos_theta = np.abs(M[0, 0])
273    sin_theta = np.abs(M[0, 1])
274    dest_w = int(w * cos_theta + h * sin_theta)
275    dest_h = int(w * sin_theta + h * cos_theta)
276
277    # adjust gap before/after move
278    M[0, 2] += (dest_w - w) / 2.0
279    M[1, 2] += (dest_h - h) / 2.0
280
281    return cv2.warpAffine(src, M, (dest_w, dest_h))

289def is_shift_invarient_for_grayscale_imgs(a: MatLike, b: MatLike) -> bool:
290    """Check whether the given grayscale iamge is shift invariant or not"""
291
292    if not (is_grayscale(a) and is_grayscale(b)):

295    a_hist = cv2.calcHist([a], [0], None, [256], [0, 256])
296    b_hist = cv2.calcHist([b], [0], None, [256], [0, 256])
297    return np.array_equal(a_hist, b_hist)

382def draw_ruled_lines(
383        a: MatLike,
384        span: int,
385        ruled_line_color: BGRA = BGRA(255, 0, 0),
386        line_chickness: int = 1,
387        line_type: LineType = LineType.LINE_TYPE_8,
388        ruled_line_type: RuledLineType = RuledLineType.RULED_LINE_LATTICE
389) -> MatLike:
390    """Draw ruled lines"""
391    cur_x = span
392    cur_y = span
393    h, w = a.shape[:2]
394    while cur_x < w or cur_y < h:
395        if cur_x < w and \
396                ruled_line_type in \
397                [RuledLineType.RULED_LINE_LATTICE, RuledLineType.RULED_LINE_ONLY_VERTICAL]:
398            draw_line(
399                a=a,
400                st=[cur_x, 0],
401                end=[cur_x, h],
402                color=ruled_line_color,
403                line_thickness=line_chickness,
404                line_type=line_type
405            )
406        if cur_y < h and \
407                ruled_line_type in \
408                [RuledLineType.RULED_LINE_LATTICE, RuledLineType.RULED_LINE_ONLY_HORIZONTAL]:
409            draw_line(
410                a=a,
411                st=[0, cur_y],
412                end=[w, cur_y],
413                color=ruled_line_color,
414                line_thickness=line_chickness,
415                line_type=line_type
416            )
417
418        cur_x = min(cur_x + span, w)
419        cur_y = min(cur_y + span, h)
420
421    return a

610def is_high_contract(
611    col1: Tuple[int, int, int],
612    col2: Tuple[int, int, int],
613    verbose: bool = False
614) -> bool:
615    """Evaluates whether the given two colors are high contrast or not"""
616    # brightness diff test formula
617    # abs((299 * R1 + 587 * G1 + 114 * B1) / 1000) -
618    # (299 * R2 + 587 * G2 + 114 * B2) / 1000) > 125
619    # hue diff test formula
620    # abs(R1 - R2) + abs(G1 - G2) + abs(B1 - B2) > 500
621    b1, g1, r1 = col1
622    b2, g2, r2 = col2
623    # brightness test
624    bd1 = (r1 * 299 + g1 * 587 + b1 * 114) // 1000
625    bd2 = (r2 * 299 + g2 * 587 + b2 * 114) // 1000
626    brightness_diff = abs(bd1 - bd2)
627    if verbose:
628        print(f"bd1={bd1},bd2={bd2},brightness_diff={brightness_diff}")
629    if brightness_diff <= 125:
630        if verbose:
631            print("brightness check failed - not high contract")
632        return False
633    hue_diff = abs(b1 - b2) + abs(g1 - g2) + abs(r1 - r2)
634    if verbose:
635        print(f"hue_diff={hue_diff}")
636    if hue_diff <= 500:
637        if verbose:
638            print("hue check failed - not high contract")
639            return False
640    print("both brightness and hue check passed. High contract")
641    return True

743    buf = np.empty(shape=[h // skip, w // skip, d])
744    h_idx = 0

747        w_idx = 0
748        for x in range(0, w, skip):
749            for d_idx in range(d):

751            w_idx += 1
752        h_idx += 1

259def affine_transform(src: MatLike, scale: int, angle: int) -> MatLike:
260    """Apply affine transformation"""
261    h, w = src.shape[:2]
262    center = (w // 2, h // 2)
263    # M is affin trans matrix
264    M = cv2.getRotationMatrix2D(center, angle, scale)
265    # M = [a    b   c]
266    #     [d    e   f]
267    # a .. scale up/down against X axis. scale * cos(angle)
268    # e .. scale up/down against Y axis. scale * sin(angle)
269    # b .. shear/rotate against x axis
270    # d .. shear/rotate against y axis
271    # c .. parallel move against x axis. (1-a)*center.x - b*center.y
272    # f .. parallel move against y axis. b*center.x - (1-a)*center.y
273    cos_theta = np.abs(M[0, 0])
